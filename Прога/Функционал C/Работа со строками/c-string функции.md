```C++
\#include <stdio.h>
\#include <string.h>
int main(void){
const char* str="abcdef123";
printf("%Iu\n",strchr(str,'c')-str); // Возвращает первый указатель на символ в строке
printf("%Iu\n",strstr(str,"def")-str);// Возвращает первый указатель на строку в строке
/Поиск символов строки str2 в строке str1.
Возвращает длину начального участка строки str1,
который состоит только из символов строки str2./
printf("%Iu\n",strspn(str,"cba"));
printf("%Iu\n",strcspn(str,"123"));// Возвращает первый указатель на либой из набора символов в строке
/---------------------------------------------
Разбиваем строку на лексемы по разделителю
-----------------------------------------------/
char  lexems[]="abc def 123 4567";
char * pch = strtok (lexems," ,.-"); // во втором параметре указаны разделитель (пробел, запятая, точка, тире)
while (pch != NULL)                         // пока есть лексемы
{
printf("%s\n",pch);
pch = strtok (NULL, " ,.-");
}
return 0;
}
```
